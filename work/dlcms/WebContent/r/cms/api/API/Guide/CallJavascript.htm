    this._ReverseMouseActionCallback=function(geometry,specialGeometry)
    {
        this.Map.ShowLoading(true);
        Ext.Ajax.request({  
                     scope :this,
                     //请求地址   
                     url: AppRelativeClientUrl+"Geocoding/LiquidGIS/Reverse.gis",   
                     //提交参数组   
                     params: {  x:specialGeometry.X,
                                y:specialGeometry.Y,
                                format:"application/json"
                              },   
                     //成功时回调   
                     success: function(response, options) {   
                        //获取响应的json字符串   
                        this.Map.ShowLoading(false);
 
                        var res = Ext.util.JSON.decode(response.responseText);   
                        Ext.MessageBox.show({
                                    title:"当前位置地址信息",
                                    msg:res.join("<br/>"),
                                    width:400,
                                    height:80,
                                    icon:Ext.MessageBox.INFO,
                                    closable:true
                                });
                               
                    }   
            });  
    }
   
    // 查询回调
    this._QueryMouseActionCallback = function(geometry,specialGeometry)
    {
         
        var locationAddressType = Ext.getCmp(this.CurrentAddressTypeID).getValue();
        var locationKey = Ext.getCmp(this.KeyID).getValue();
        var wkt = typeof(geometry.GetWKT)=="undefined"?null:geometry.GetWKT();
        var buffer = typeof(geometry.GetWKT)=="undefined"?0:geometry.Buffer;
 
 
        if(locationKey==""&&wkt==null)
        {
            Ext.MessageBox.alert("警告","请输入查询条件！");
            return;
        }
        this.Map.ShowLoading(true);
        Ext.Ajax.request({  
                     scope :this,
                     //请求地址   
                     url: AppRelativeClientUrl+"Geocoding/LiquidGIS/FindAddress.gis",   
                     //提交参数组   
                     params: {  wkt:wkt,
                                buffer:buffer,
                                key:locationKey,hasGeometry:false,
                                spatialRelationship:"Intersects",
                                format:"application/json",
                                addressType:locationAddressType
                              },   
                     //成功时回调   
                     success: function(response, options) {   
                        //获取响应的json字符串   
                        this.Map.ShowLoading(false);
                        var res = Ext.util.JSON.decode(response.responseText);   
                        this.QueryAddress = res;
                        this.QueryDataTable = this.ConvertAddressToDataTable(res);
                        this.ShowResultData();
                               
                    }   
            });  
         
         
    } 
     
    this.ConvertAddressToDataTable=function(address)
    {
        var table = {};
        table.TotalPage = 1;
        table.TotalCount  = address.length;
        table.Rows = {};
        table.Columns ={};
        table.Columns.DataColumns = [];
        table.Columns.DataColumns[0] = {ColumnName:"Type",AliasName: "类型"};
        table.Columns.DataColumns[1] = {ColumnName:"Name",AliasName: "地址"};
         
        //table.Columns.DataColumns[1] = {ColumnName:"GeoGraphicalName",AliasName: "标准地名"};
         
        table.Rows.DataRows = [];
        for(var i=0;i<address.length;i++)
        {
            var row = {};
            row.Value = [address[i].Type , address[i].Name/*,address[i].AliasName*/];
            row.ObjectID  = address[i].ObjectID;
            row.LayerName   = address[i].LayerName ;
            row.Envelope   = address[i].Envelope ;
            row.ICON = AppRelativeClientUrl+"Images/Google/Letter/S1.gif";
            table.Rows.DataRows[i] = row;
        }
         
        return table;
    }

        return new Ext.grid.ColumnModel(column);

    this.ConvertDataTableToExtStore = function(table)

        function GetArraryData(table)
        {
            var data = new Array();
            if(table==null||table.Rows==null||table.Rows.DataRows==null)return null;
             
            for(var i=0;i<table.Rows.DataRows.length;i++)
            {
                var value = table.Rows.DataRows[i].Value;
 
                for(var k=0;k<value.length;k++)
                {
                    if(value[k] == ""|| value[k]==null)  value[k] = "";
                }
 
                // 自动补全ObjectID
                value.push(table.Rows.DataRows[i].ObjectID);
                // 设置记录集的DataRow，便于以后查找
                value.push(table.Rows.DataRows[i]);            
                data.push(value) ;
            }
             
            return data;
        }
         
         
        function GetArrayReader(table)
        {
            var reader = new Array();
            if(table==null||table.Columns==null||table.Columns.DataColumns==null)return null;
            for(var i=0;i<table.Columns.DataColumns.length;i++)
            {
                reader.push({name:table.Columns.DataColumns[i].ColumnName});
            }
            reader.push("ObjectID");
            // 设置记录集的DataRow
            reader.push("DataRow");
             
            return new Ext.data.ArrayReader({},reader);
             
        }
         
     
        return new Ext.data.Store({
                reader: GetArrayReader(table),
                data: GetArraryData(table)
            });
             
    }

    {
        var tag = event.srcElement.tagName;
        var typeName = event.srcElement.TypeName;
        var record = grid.getStore().getAt(rowIndex);
        var objectID = record.get("ObjectID");
        var dataRow = record.get("DataRow");
        this.__objectID = objectID;
        this.__record = record;
         
        if(tag == "IMG")
        {
            var range = dataRow.Envelope;
            OpenWebGIS.LiquidGIS.Map.Tools.IdentifyPropertyGrid.prototype.GoToRange(this.Map,range,dataRow.LayerID,[objectID]);        
        }
        else if(tag == "A")
        {
            // 弹出页面
            if(this.OpenInfo!=null) this.OpenInfo(objectID,dataRow);
        }
         
    }
